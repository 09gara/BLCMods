        set GD_Flax_Items.Item.Item_Candy_Red bPlayerUseItemOnPickup False

        set GD_Flax_Items.Item.Item_Candy_Blue bPlayerUseItemOnPickup False

        set GD_Flax_Items.Item.Item_Wisp bPlayerUseItemOnPickup False

    #</Code>
