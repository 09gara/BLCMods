#<Anarchyless Gaige>

    By Koby

    Sets Smaller, Lighter, Faster's maximum level to 5 so you can go down the Ordered Chaos skill tree without having to pick Anarchy.

    Goes well with my TPS Gaige mod.

    Merge with other hotfixed mods as usually.

    #<Code>

        #<hotfix><key>"SparkOnDemandPatchEntry-AnarchylessGaige1"</key><value>"GD_Tulip_Mechro_Streaming,GD_Tulip_Mechromancer_Skills.EmbraceChaos.SmallerLighterFaster,MaxGrade,,5"</value><on>

    #</Code>

#</Anarchyless Gaige>

set Transient.SparkServiceConfiguration_6 Keys ("SparkOnDemandPatchEntry-AnarchylessGaige1")

set Transient.SparkServiceConfiguration_6 Values ("GD_Tulip_Mechro_Streaming,GD_Tulip_Mechromancer_Skills.EmbraceChaos.SmallerLighterFaster,MaxGrade,,5")
